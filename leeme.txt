    <!-- Vista de Login -->
    <div id="loginView" class="container">
        <h1>Iniciar Sesión</h1>
        <form id="loginForm">
            <div class="form-group">
                <label for="email">Correo Electrónico</label>
                <input type="email" id="email" name="email" required>
            </div>
            <div class="form-group">
                <label for="password">Contraseña</label>
                <input type="password" id="password" name="password" required>
            </div>
            <button type="submit" id="loginButton">Iniciar Sesión</button>
            <div class="error-message" id="errorMessage">Error en el inicio de sesión</div>
        </form>
        <div class="loading" id="loginLoading">
            <div class="spinner"></div>
            <p>Iniciando sesión...</p>
        </div>
    </div>

    <!-- Vista del QR -->
    <div id="qrView" class="container hidden">
        <h1>Escanear Código QR</h1>
        <p>Utiliza tu aplicación de WhatsApp para escanear el siguiente código:</p>
        <div class="qr-container">
            <img id="qrImage" src="" alt="Código QR">
        </div>
        <div class="loading" id="qrLoading">
            <div class="spinner"></div>
            <p>Cargando código QR...</p>
        </div>
        <button id="backButton" style="margin-top: 20px;">Volver al Login</button>
    </div>

    <script>
        // Elementos del DOM
        const loginView = document.getElementById('loginView');
        const qrView = document.getElementById('qrView');
        const loginForm = document.getElementById('loginForm');
        const loginButton = document.getElementById('loginButton');
        const errorMessage = document.getElementById('errorMessage');
        const loginLoading = document.getElementById('loginLoading');
        const qrImage = document.getElementById('qrImage');
        const qrLoading = document.getElementById('qrLoading');
        const backButton = document.getElementById('backButton');

        // Manejar el envío del formulario de login
        loginForm.addEventListener('submit', async (e) => {
            e.preventDefault();
            
            // Obtener datos del formulario
            const email = document.getElementById('email').value;
            const password = document.getElementById('password').value;
            
            // Mostrar indicador de carga
            loginButton.disabled = true;
            loginLoading.style.display = 'block';
            errorMessage.style.display = 'none';
            
            try {
                // Realizar solicitud a la API
                const response = await fetch('http://127.0.0.1:8000/api/login', {
                    method: 'POST',
                    headers: {
                        "Content-Type": "application/json",
                        "Accept": "application/json"
                    },
                    body: JSON.stringify({ email, password })
                });
                
                // Verificar si la respuesta es exitosa
                if (response.ok) {
                    const data = await response.json();
                    
                    // Extraer y mostrar el código QR
                    if (data.whatsapp_api_response && data.whatsapp_api_response.base64) {
                        // Cambiar a la vista del QR
                        loginView.classList.add('hidden');
                        qrView.classList.remove('hidden');
                        
                        // Mostrar el QR
                        qrImage.src = data.whatsapp_api_response.base64;
                        qrLoading.style.display = 'none';
                    } else {
                        throw new Error('No se recibió el código QR');
                    }
                } else {
                    throw new Error('Credenciales incorrectas');
                }
            } catch (error) {
                console.error('Error en el login:', error);
                errorMessage.textContent = error.message || 'Error en el inicio de sesión';
                errorMessage.style.display = 'block';
            } finally {
                // Restaurar estado del formulario
                loginButton.disabled = false;
                loginLoading.style.display = 'none';
            }
        });

        // Manejar el botón de volver al login
        backButton.addEventListener('click', () => {
            qrView.classList.add('hidden');
            loginView.classList.remove('hidden');
            loginForm.reset();
        });
    </script>
